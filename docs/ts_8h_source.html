363/* Context structure for the generic verify method. */
364
365/* Verify the signer's certificate and the signature of the response. */
366# define TS_VFY_SIGNATURE (1u << 0)
367/* Verify the version number of the response. */
368# define TS_VFY_VERSION (1u << 1)
369/* Verify if the policy supplied by the user matches the policy of the TSA. */
370# define TS_VFY_POLICY (1u << 2)
371/*
372 * Verify the message imprint provided by the user. This flag should not be
373 * specified with TS_VFY_DATA.
374 */
375# define TS_VFY_IMPRINT (1u << 3)
376/*
377 * Verify the message imprint computed by the verify method from the user
378 * provided data and the MD algorithm of the response. This flag should not
379 * be specified with TS_VFY_IMPRINT.
380 */
381# define TS_VFY_DATA (1u << 4)
382/* Verify the nonce value. */
383# define TS_VFY_NONCE (1u << 5)
384/* Verify if the TSA name field matches the signer certificate. */
385# define TS_VFY_SIGNER (1u << 6)
386/* Verify if the TSA name field equals to the user provided name. */
387# define TS_VFY_TSA_NAME (1u << 7)
388
389/* You can use the following convenience constants. */
390# define TS_VFY_ALL_IMPRINT (TS_VFY_SIGNATURE \
391 | TS_VFY_VERSION \
392 | TS_VFY_POLICY \
393 | TS_VFY_IMPRINT \
394 | TS_VFY_NONCE \
395 | TS_VFY_SIGNER \
396 | TS_VFY_TSA_NAME)
397# define TS_VFY_ALL_DATA (TS_VFY_SIGNATURE \
398 | TS_VFY_VERSION \
399 | TS_VFY_POLICY \
400 | TS_VFY_DATA \
401 | TS_VFY_NONCE \
402 | TS_VFY_SIGNER \
403 | TS_VFY_TSA_NAME)
