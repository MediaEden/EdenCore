388/* Used to pass pictures into the encoder, and to get picture data back out of
389 * the encoder. The input and output semantics are different */
390typedef struct x265_picture
391{
392 /* presentation time stamp: user-specified, returned on output */
393 int64_t pts;
394
395 /* display time stamp: ignored on input, copied from reordered pts. Returned
396 * on output */
397 int64_t dts;
398
399 /* force quantizer for != X265_QP_AUTO */
400 /* The value provided on input is returned with the same picture (POC) on
401 * output */
402 void* userData;
403
404 /* Must be specified on input pictures, the number of planes is determined
405 * by the colorSpace value */
406 void* planes[3];
407
408 /* Stride is the number of bytes between row starts */
409 int stride[3];
410
411 /* Must be specified on input pictures. x265_picture_init() will set it to
412 * the encoder's internal bit depth, but this field must describe the depth
413 * of the input pictures. Must be between 8 and 16. Values larger than 8
414 * imply 16bits per input sample. If input bit depth is larger than the
415 * internal bit depth, the encoder will down-shift pixels. Input samples
416 * larger than 8bits will be masked to internal bit depth. On output the
417 * bitDepth will be the internal encoder bit depth */
418 int bitDepth;
419
420 /* Must be specified on input pictures: X265_TYPE_AUTO or other.
421 * x265_picture_init() sets this to auto, returned on output */
422 int sliceType;
423
424 /* Ignored on input, set to picture count, returned on output */
425 int poc;
426
427 /* Must be specified on input pictures: X265_CSP_I420 or other. It must
428 * match the internal color space of the encoder. x265_picture_init() will
429 * initialize this value to the internal color space */
430 int colorSpace;
431
432 /* Force the slice base QP for this picture within the encoder. Set to 0
433 * to allow the encoder to determine base QP */
434 int forceqp;
435
436 /* If param.analysisLoad and param.analysisSave are disabled, this field is
437 * ignored on input and output. Else the user must call x265_alloc_analysis_data()
438 * to allocate analysis buffers for every picture passed to the encoder.
439 *
440 * On input when param.analysisLoad is enabled and analysisData
441 * member pointers are valid, the encoder will use the data stored here to
442 * reduce encoder work.
443 *
444 * On output when param.analysisSave is enabled and analysisData
445 * member pointers are valid, the encoder will write output analysis into
446 * this data structure */
447 x265_analysis_data analysisData;
448
449 /* An array of quantizer offsets to be applied to this image during encoding.
450 * These are added on top of the decisions made by rateControl.
451 * Adaptive quantization must be enabled to use this feature. These quantizer
452 * offsets should be given for each 16x16 block (8x8 block, when qg-size is 8).
453 * Behavior if quant offsets differ between encoding passes is undefined. */
454 float *quantOffsets;

458
459 /* User defined SEI */
460 x265_sei userSEI;
461
462 /* Ratecontrol statistics for collecting the ratecontrol information.
463 * It is not used for collecting the last pass ratecontrol data in
464 * multi pass ratecontrol mode. */
465 void* rcData;
466
467 size_t framesize;
468
469 int height;
470
471 // pts is reordered in the order of encoding.
472 int64_t reorderedPts;
473
474 //Dolby Vision RPU metadata
475 x265_dolby_vision_rpu rpu;
476
477 int fieldNum;
478
479 //SEI picture structure message
480 uint32_t picStruct;
481
482 int width;
483} x265_picture;

509#define X265_CPU_XOP (1 << 10) /* AMD XOP */
510#define X265_CPU_FMA4 (1 << 11) /* AMD FMA4 */
511#define X265_CPU_FMA3 (1 << 12) /* Intel FMA3 */
512#define X265_CPU_BMI1 (1 << 13) /* BMI1 */

516/* x86 modifiers */
517#define X265_CPU_CACHELINE_32 (1 << 17) /* avoid memory loads that span the border between two cachelines */
518#define X265_CPU_CACHELINE_64 (1 << 18) /* 32/64 is the size of a cacheline in bytes */
519#define X265_CPU_SSE2_IS_SLOW (1 << 19) /* avoid most SSE2 functions on Athlon64 */
520#define X265_CPU_SSE2_IS_FAST (1 << 20) /* a few functions are only faster on Core2 and Phenom */
521#define X265_CPU_SLOW_SHUFFLE (1 << 21) /* The Conroe has a slow shuffle unit (relative to overall SSE performance) */
522#define X265_CPU_STACK_MOD4 (1 << 22) /* if stack is only mod4 and not mod16 */
523#define X265_CPU_SLOW_ATOM (1 << 23) /* The Atom is terrible: slow SSE unaligned loads, slow
524 * SIMD multiplies, slow SIMD variable shifts, slow pshufb,
525 * cacheline split penalties -- gather everything here that
526 * isn't shared by other CPUs to avoid making half a dozen
527 * new SLOW flags. */
528#define X265_CPU_SLOW_PSHUFB (1 << 24) /* such as on the Intel Atom */
529#define X265_CPU_SLOW_PALIGNR (1 << 25) /* such as on the AMD Bobcat */
