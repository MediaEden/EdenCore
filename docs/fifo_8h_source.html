169static inline uint8_t *av_fifo_peek2(const AVFifoBuffer *f, int offs)
170{
171 uint8_t *ptr = f->rptr + offs;
172 if (ptr >= f->end)
173 ptr = f->buffer + (ptr - f->end);
174 else if (ptr < f->buffer)
175 ptr = f->end - (f->buffer - ptr);
176 return ptr;
177}
